data = data.drop(columns=['Country_code', 'Cumulative_cases', 'Cumulative_deaths']) 
#  
data.rename(columns={'Country': '', 'WHO_region': '', 'New_cases': '_', 'New_deaths': ''}, inplace=True)
#    ,    
data['Date_reported'] = pd.to_datetime(data['Date_reported'], format='%Y-%m-%d')
data[''] = data['Date_reported'].apply(lambda x: x.month)
data[''] = data['Date_reported'].apply(lambda x: x.year)
data['_'] = data[''].astype(str) + '-' + data[''].astype(str)
data['_'] = pd.to_datetime(data['_'], format='%Y-%m').dt.date
