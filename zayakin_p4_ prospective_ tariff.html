def total_monthly_account(row):
    
#  :
    account_minutes = 0 #     
    account_sms = 0 #   SMS  
    account_traffic = 0 #   -  
    minutes = row['calls_duration'] #   
    sms = row['messages_count'] #   SMS
    traffic = row['traffic_volume'] #  
    tariff = row['tariff'] #  
    limit_minutes = row['minutes_included'] #       
    limit_sms = row['messages_included'] #   SMS     
    limit_traffic = row['mb_per_month_included'] #        
    rub_monthly_fee = row['rub_monthly_fee'] #    
    rub_per_gb = row['rub_per_gb'] #      
    rub_per_message = row['rub_per_message'] #  SMS   
    rub_per_minute = row['rub_per_minute'] #      

#      , ,     .
#     ,    -     
#   -        .
    
    if minutes > limit_minutes:
        account_minutes += (minutes - limit_minutes) * rub_per_minute
    if minutes < limit_minutes:
        account_minutes = 0
    if sms > limit_sms:
        account_sms += (sms - limit_sms) * rub_per_message
    if sms < limit_sms:
        account_sms = 0
    if traffic > limit_traffic:
        account_traffic += (np.ceil((traffic - limit_traffic) / 1024 )) * rub_per_gb #     
    if traffic < limit_traffic:
        account_traffic = 0
        
#        ,
#              
    return account_minutes + account_sms + account_traffic + rub_monthly_fee
